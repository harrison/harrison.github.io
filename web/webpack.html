                      # 打包图片资源

                      limit: 可以规定图片大小限制，在limit限制以内的图片会被转化为base64存储 esModule: 在url-loader升级3.0.0后，默认情况下，url-loader生成使用ES module语法的js模块，这就导致生成后的dom中img标签的src属性变为obj对象，无法正常显示，所以需要将esModule属性设置为false，生成CommonJs语法 name: 可以规定生成后的图片文件名，[hash: 10] 代表取文件哈希值前10位，[ext] 是文件扩展名

                      {
  test: /\.(jpg|png|gif)/,
  loader: 'url-loader',
  options: {
    limit: 8 * 1024,
    esModule: false,
    name: '[hash:10].[ext]'
  }
}

"url-loader": "^3.0.0",

                      # css兼容性处理

                      需要在package.json中定义browserslist，规定兼容哪些浏览器 默认为生产环境，如需切换到开发环境需要设置process.env.NODE_ENV环境变量

                      process.env.NODE_ENV = 'production'


"browserslist": {
  "development": [
    "last 1 chrome version",
    "last 1 firefox version",
    "last 1 safari version"
  ],
  "production": [
    ">0.2%",
    "not dead",
    "not op_mini all"
  ]
}


{
  test: /\.css$/,
  use: [
    // 'style-loader',
    MiniCssExtractPlugin.loader,
    'css-loader',
    {
      loader: 'postcss-loader',
      options: {
        ident: 'postcss',
        plugins: () => [
          require('postcss-preset-env')()
        ]
      }
    }
  ]
}


  "postcss-loader": "^3.0.0",
  "postcss-preset-env": "^6.7.0",
